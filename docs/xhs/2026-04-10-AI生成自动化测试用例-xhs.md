```python
import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB

data = {
    'test_case': ['登录功能测试', '注册功能测试', '支付功能测试', '订单功能测试'],
    'function': ['登录', '注册', '支付', '订单']
}
df = pd.DataFrame(data)

vectorizer = CountVectorizer()
X = vectorizer.fit_transform(df['test_case'])

model = MultinomialNB()
model.fit(X, df['function'])

new_test_case = '查询功能测试'
new_test_case_vector = vectorizer.transform([new_test_case])
predicted_function = model.predict(new_test_case_vector)

print(f'预测的功能点：{predicted_function[0]}')
```
